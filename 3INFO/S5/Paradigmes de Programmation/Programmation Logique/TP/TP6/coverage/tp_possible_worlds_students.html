% dana likes cody
% bess does not like dana
% cody does not like abby
% nobody likes someone who does not like her
% abby likes everyone who likes bess
% dana likes everyone bess likes
% everybody likes somebody

people([abby, bess, cody, dana]) 65537 .

% 1.1

addAll(_2571, [], R, R) 4 .
addAll(E, [A|L], M, R) :-
         16 addAll(E, L, [likes(E, A)|M], R) 16 .

mkpairs(_3575, [], R, R) 1 .
mkpairs(L, [E|Li], A, R) :-
         4 addAll(E, L, A, A1),
         4 mkpairs(L, Li, A1, R) 4 .

make_all_pairs(L, R) :-
         1 mkpairs(L, L, [], R) 1 .


% 1.2

sub_list([], []) 65536 .
sub_list([_5747|L], R) :-
         65535 sub_list(L, R) 524288 .
sub_list([A|L], [A|R]) :-
         65535 sub_list(L, R) 524288 .


% 1.3

         31404 ! 31404 .

         250128 remove(E, L, R) 250128 .

proposition1(L) :-
         50625 contains(likes(dana, cody), L) 27000 .

proposition2(L) :-
         27000 contains_not(likes(bess, dana), L) 12600 .

         12600 contains_not(likes(cody, abby), L) 5880 .

         4368 remove(likes(B, A), L, L1),
         4368 proposition4(L1) 276 .

aux6(_16804, []) 4 .

aux7([], _19008) :-

% 1.4

possible_worlds(W) :-
         1 people(P),
         1 make_all_pairs(P, L),
         1 sub_list(L, W),
         65536 proposition7(W),
         50625 proposition1(W),
         27000 proposition2(W),
         12600 proposition3(W),
         5880 proposition4(W),
         100 proposition5(W),
         16 proposition6(W) 4 .
